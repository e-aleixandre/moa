He creado el CRUD completo. Aquí tienes un ejemplo de cómo usarlo: 
package main

import "net/http"

func main() {
	r := chi.NewRouter()
	r.Use(middleware.Logger)

	r.Route("/users", func(r chi.Router) {
		r.Post("/", handlers.HandleCreateUser)
		r.Get("/{id}", handlers.HandleGetUser)
	})

	http.ListenAndServe(":3000", r) // TODO: handle error
}
