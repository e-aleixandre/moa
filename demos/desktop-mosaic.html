Done. NewFileStore now takes (baseDir, cwd) and scopes sessions under <basename>_<hash12>. The hash ensures uniqueness while keeping dirs human-readable.

gofunc scopeKey(cwd string) string {
    clean := filepath.Clean(cwd)
    base := filepath.Base(clean)
    h := sha256.Sum256([]byte(clean))
    return base + "_" + hex.EncodeToString(h[:6])
}
